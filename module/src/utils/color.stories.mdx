## Color

Utilities for working with colors.

I was forced to spell it using American English... I'm not happy about it either.

### hexToRGB

Convert a hex string to an RGB object

```ts
function hexToRGB(hex: string): IRGBColor;
```

```ts
import { hexToRGB } from '@rocketmakers/armstrong';

const result = hexToRGB("#FFFFFF");

// result
{
    red: 255,
    blue: 255,
    green: 255
}

const result = hexToRGB("#FFFFFF00");

// result
{
    red: 255,
    blue: 255,
    green: 255,
    alpha: 0
}
```

### colorToRGB

Ensure that a Color (a hex string or an RGB object) is an RGB object, converting it if not

This is useful for components which will expect either, but you only want to work with RGB

```ts
function colorToRGB(color: Color): IRGBColor;
```

```ts
import { colorToRGB } from '@rocketmakers/armstrong';

const result = colorToRGB("#FFFFFF");

// result
{
    red: 255,
    blue: 255,
    green: 255
}

const result = colorToRGB({ red: 0, green: 0, blue: 0 });

// result
{
    red: 0,
    blue: 0,
    green: 0
}
```

### RGBToHex

Convert an RGB string to a hex colour

```ts
function RGBToHex({ red, green, blue, alpha }: IRGBColor): string;
```

```ts
import { RGBToHex } from "@rocketmakers/armstrong";

const result = RGBToHex({ red: 255, green: 255, blue: 255 });

// result
("#FFFFFF");

const result = RGBToHex({ red: 255, green: 255, blue: 255, alpha: 0.01 });

// result
("#FFFFFF03");
```

### lerpRGB

Get a colour lerped using a progress between a start and end colour

```ts
function lerpRGB(
  startColor: IRGBColor,
  endColor: IRGBColor,
  /** out of 100 */ progress: number
): IRGBColor;
```

```ts
import { lerpRGB } from '@rocketmakers/armstrong';

const startColor = { red: 0, green: 255, blue: 0 };
const endColor = { red: 255, green: 0, blue: 0 };

const result = lerpRGB(startColor, endColor, 50)

// result
{
    red: 127,
    green: 128,
    blue: 0
}
```

### multiLerpRGB

Get a colour lerped between multiple breakpoints

```ts
function multiLerpRGB(
  colors: IRGBColor[],
  /** out of 100 */ progress: number
): IRGBColor;
```

```ts
import { multiLerpRGB } from '@rocketmakers/armstrong';

const startColor = { red: 0, green: 255, blue: 0 };
const midColor = { red: 255, green: 0, blue: 0 };
const endColor = { red: 0, green: 0, blue: 255 };

const result = multiLerpRGB([startColor, midColor, endColor], 25)

// result
{
    red: 127,
    green: 127,
    blue: 0
}

const result = multiLerpRGB([startColor, midColor, endColor], 50)

// result
{
    red: 255,
    green: 0,
    blue: 0
}
```

### lerpHex

Get a colour lerped between a start and end colour as a hex

```ts
function lerpHex(
  startColor: string,
  endColor: string,
  /** out of 100 */ progress: number
): string;
```

```ts
import { lerpHex } from "@rocketmakers/armstrong";

const startColor = "#00FF00";
const endColor = "#FF0000";

const result = lerpHex(startColor, endColor, 50);

// result
("#7F7F00");
```

### colorToRGBCSString

Turn an rgb object into a css string, i.e. rgb(100, 100, 90)

```ts
function colorToRGBCSString(color: IRGBColor);
```

```ts
import { colorToRGBCSString } from "@rocketmakers/armstrong";

const result = colorToRGBCSString({ red: 0, green: 100, blue: 200 });

// result
("rgb(0, 100, 200)");

const result = colorToRGBCSString({ red: 0, green: 100, blue: 200, alpha: 1 });

// result
("rgba(0, 100, 200, 1)");
```
